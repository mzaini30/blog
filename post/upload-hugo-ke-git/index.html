Upload Hugo Ke Git

September 26, 2018 
upload_git(){
	git status
	git add -A .
	git commit -m 'oke'
	git push
}

while true
	do

		# upload dulu ke source blog

		upload_git

		# deploy

		hugo

		# masuk ke folder public

		cd public

		# upload lagi

		upload_git

		# keluar dari folder

		cd ..

		# untuk pause

		read -rsp $'Tekan sembarang tombol untuk upload lagi.\n\n\t'"blog"$'\n\n' -n 1 key

	done
