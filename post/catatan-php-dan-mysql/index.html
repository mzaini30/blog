Catatan Mysql

October 02, 2018

Melacak lokasi MySQL

which mysql

Export database

mysqldump -u username -p nama_database > psikoweb.sql

Masuk ke MySQL

mysql -u username -p

Menampilkan database

show databases;

Membuat database

create database psikoweb;

Memilih database

use psikoweb;

Membuat table

create table operator(
	id int (20) not null auto_increment,
	nama varchar (20) not null,
	updated timestamp,
	primary key (id)
);

Menampilkan table

show tables;

Menghapus table

drop table kucing;

Menghapus isi tabel

truncate table kucing;

Menghapus database

drop database bioskop;

Menampilkan form table

desc peserta;

Menambah Kolom Baru

alter table nama_tabel add nama_kolom varchar (50) not null;

Menghapus kolom

alter table kucing drop nama;

Insert data

insert into kucing (nama, asal, kewarganegaraan) values ('anggora', 'jakarta', 'eropa');
